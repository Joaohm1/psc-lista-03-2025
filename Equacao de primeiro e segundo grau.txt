import java.util.Scanner;

public class Main
{
	public static void main(String[] args) {
	
	Scanner scanner = new Scanner(System.in);
	
	
	System.out.println("Digite valor de a :");
	double a = scanner.nextDouble();
	
	System.out.println("Digite valor de b :");
	double b = scanner.nextDouble();
	
	System.out.println("Digite valor de c :");
	double c = scanner.nextDouble();
	
	if(a == 0 && b == 0 && c != 0 ){
	    System.out.println("Coeficientes informados incorretamente.");
	    return;
	}
	
	if(a == 0 && b != 0 ){
	    System.out.println("Essa é uma equação de primeiro grau");
	    double x = - c / b;
	    System.out.println("Raiz real: x = " +  x);
	    return;
	}
	 // A partir daqui: equação de 2º grau
        double delta = b*b - 4*a*c;
        
        if(delta < 0 ){
             System.out.println("Essa equação não possui raízes reais .");
}
	else if(delta == 0){
	    double x = -b / (2*a);
	    System.out.println("Esta equacao possui duas raizes reais iguais.");
	    System.out.println("x1 = x2 =" + x );
	    
	    
	}else{
	    double sqrt = Math.sqrt(delta);
	      double x1 = (-b + sqrt) / (2*a);
            double x2 = (-b - sqrt) / (2*a);
            System.out.println("Esta equação possui duas raízes reais diferentes.");
            System.out.printf("x1 = %.6f, x2 = %.6f%n", x1, x2);
	}
	}
	
	
	}
